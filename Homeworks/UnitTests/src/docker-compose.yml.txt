version: '3.8'

services:
  webapi:
    build:
      context: .
      dockerfile: Dockerfile
    ports:
      - "8080:8080"
    environment:
      - ASPNETCORE_ENVIRONMENT=Development
      # если нужно явно:
      # - ASPNETCORE_HTTP_PORTS=8080
      - ConnectionStrings__PromoCodeFactoryDb=Host=db;Port=5432;Database=promocode_factory_db;Username=postgres;Password=111111;
    depends_on:
      db:
        condition: service_healthy
    restart: unless-stopped

  db:
    image: postgres:16
    environment:
      - POSTGRES_DB=promocode_factory_db
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=111111
      # УДАЛЕНО: POSTGRES_HOST_AUTH_METHOD=trust (конфликтует с паролем)
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres -d promocode_factory_db"]
      interval: 10s
      timeout: 5s
      retries: 5

volumes:
